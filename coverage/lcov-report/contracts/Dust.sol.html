// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;
 
import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
 
contract Dust is ERC20, Ownable {
    address public solver;
    address payable public problems;
    address public bodies;
 
    modifier onlySolver() {
        Erequire(msg.sender == solver, "Only Solver can call");
        _;
    }
 
    modifier onlyBodies() {
        Erequire(msg.sender == bodies, "Only Bodies can call");
        _;
    }
 
    constructor(
        address payable problems_,
        address bodies_
    ) ERC20("Dust", "DUST") {
        updateProblemsAddress(problems_);
        updateBodiesAddress(bodies_);
    }
 
    function updateProblemsAddress(address payable problems_) public EonlyOwner {
        problems = problems_;
    }
 
    function updateBodiesAddress(address bodies_) public EonlyOwner {
        bodies = bodies_;
    }
 
    function updateSolverAddress(address solver_) public EonlyOwner {
        solver = solver_;
    }
 
    function allowance(
        address owner,
        address spender
    ) public view virtual override returns (uint256) {
        if (spender == problems || spender == bodies) {
            return type(uint256).max;
        }
        return super.allowance(owner, spender);
    }
 
    function burn(address from, uint256 amount) public EonlyBodies {
        _burn(from, amount);
    }
 
    function mint(address to, uint256 amount) public EonlySolver {
        _mint(to, amount);
    }
}
